#include <iostream>
#include <ctime>
using namespace std;

int main()
{
	setlocale(LC_ALL, "Rus");
	srand(time(NULL));
	int mas[50];
	for (int i = 0; i < 50; i++)
	{
		mas[i] = rand() % 50;
	}
	cout << "Массив: " << endl;
	for (int i = 0; i < 50; i++)
	{
		cout << mas[i] << " " << endl;
	}
	int min = mas[0];
	int a = 1;
	for (int i = 0; i < 50; i++)
	{
		if (mas[i] < min)
		{
			min = mas[i];
			a = i;
		}
	}
	cout << "Наименьшее >> " << mas[a] << endl;
	a = a + 1;
	cout << "Его место >> " << a << endl;
	




	system("pause");

}
