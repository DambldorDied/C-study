void print(int* m, int size)

void sort(int* m, int size, bool f)
{
	if (f == false)
	{
		for (int i = 0; i < size - 1; i++)
		{
			for (int j = 0; j < size - 1; j++)
			{
				if (m[j] > m[j + 1])
				{
					swap(m[j], m[j + 1]);
				}
			}
		}
	}
	if (f == true)
	{
		for (int i = 0; i < size - 1; i++)
		{
			for (int j = 0; j < size - 1; j++)
			{
				if (m[j] < m[j + 1])
				{
					swap(m[j], m[j + 1]);
				}
			}
		}
	}
}

int max(int* m, int size)
{
	int max = m[0];
	for (int j = 0; j < size; j++)
	{
		for (int i = 0; i < size - 1; i++)
		{
			if (m[i] < m[i + 1])
			{
				max = m[i + 1];
			}
		}
	}
	return max;
}
