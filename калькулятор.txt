#include <iostream> 

using namespace std;

int main()
{
	setlocale(LC_ALL, "Rus");
	int n;
	float a;
	float b;
	cout << "Введите 1-ое число:";
	cin >> a;
	cout << "Введите 2-ое число:";
	cin >> b;
	cout << "Введите номер действия " << endl;
	cout<<"1-сложение-а+б; 2-вычитание-а-б; 3-умножение-аб; 4-деление-а/б:";
	cin >> n;
	switch (n)
	{
	case 1:
	{
		cout << a << "+" << b << "=" << a + b << endl;
		break;
	}
	case 2:
	{
		cout << a << "-" << b << "=" << a - b << endl;
		break;
	}
	case 3:
	{
		cout << a << "*" << b << "=" << a * b << endl;
		break;
	}
	case 4:
	{
		cout << a << "/" << b << "=" << a / b << endl;
		break;
	}
	default:
	{
		cout << "Нет такого действия!!!!!!" << endl;
	}

	}


	system("pause");
}
