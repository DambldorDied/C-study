#include <iostream>
#include <ctime>

using namespace std;

void in(int* m, int size)
{
	for (int i = 0; i < size; i++)
	{
		m[i] = rand() % 100 - 50;
	}
}

int max(int* m, int size)
{
	int max = m[0];
	for (int j = 0; j < size; j++)
	{
		for (int i = 0; i < size - 1; i++)
		{
			if (m[i] < m[i + 1])
			{
				max = i + 2;
			}
		}
	}
	return max;
}

void print(int* m, int size)
{
	for (int i = 0; i < size; i++)
	{
		cout << m[i] << " ";
	}
	cout << endl;
}

void end()
{
	cout << "\ERROR/" << endl;
	exit(0);
}
