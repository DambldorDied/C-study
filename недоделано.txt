#include <iostream>
#include <ctime>

using namespace std;

int main()
{
	setlocale(LC_ALL, "Rus");
	srand(time(NULL));
	const int n = 5;
	const int m = 7;
	int mas[n][m];
	int sum2[n];
	int sum = 0;
	int count[n];
	int count2 = 0;
	for (int i = 0; i < n; i++)
	{
		sum2[i] = 0;
		count[i] = 0;
	}
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < m; j++)
		{
			mas[i][j] = rand() % 5;
			cout << mas[i][j] << " ";
			sum = sum + mas[i][j];
			sum2[i] = sum2[i] + mas[i][j];
		}
		cout << endl;
	}
	for (int i = 0; i < n - 1; i++)
	{
		for (int j = 0; j < m; j++)
		{
			if (mas[i][j] == mas[i][j+1])
			{
				count2++;
			}
			else
			{
				if (count[i] < count2)
				{
					count[i] = count2;
					count2 = 0;
				}
				else
				{
					count2 = 0;
				}
			}
		}
	}
	int mest = 0;
	for (int i = 0; i < n - 1; i++)
	{
		if (count[i] > count[i + 1])
		{
			mest = i;
		}
		else
		{
			mest = i + 1;
		}
	}
	cout << "Сумма: " << sum << endl;
	cout << "Сумма строк:" << endl;
	for (int i = 0; i < n; i++)
	{
		cout << ">> " << sum2[i] << endl;
	}
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < m; j++)
		{
			if (mas[i][j] < mas[i + 1][j] && mas[i][j] < mas[i - 1][j] && j + 1 != m && j != 0 && i + 1 != n && i != 0 && mas[i][j] < mas[i][j + 1] && mas[i][j] < mas[i][j - 1] && mas[i][j] < mas[i + 1][j + 1] && mas[i][j] < mas[i - 1][j - 1])
			{
				mas[i][j] = -1;
				cout << mas[i][j] << " ";
			}
			else
			{
				cout << mas[i][j] << " ";
			}
		}
		cout << endl;
	}
	cout << "Номер строки, у которого кол-во подряд повторяющихся элементов больше: " << mest + 1 << endl;
	


	system("pause");
}
